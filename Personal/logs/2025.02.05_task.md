    __device__ inline bool barrier_red_or(bool vote, int name, int nThreads) {
      int ans;
      asm("{ .reg .pred p;"
          "  setp.ne.s32 p, %1, 0;"
          "  barrier.red.or.pred p, %2, %3, p; "
          "  selp.s32 %0, 1, 0, p; }"
