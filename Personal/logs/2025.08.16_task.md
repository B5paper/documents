* [v] 调研 password 的星号掩码

    ```bash
    #!/bin/bash

    stty -echo  # 关闭回显
    unset password
    prompt="Enter password: "
    while IFS= read -p "$prompt" -r -s -n1 char; do
        if [[ $char == $'\0' ]]; then  # 回车键结束
            break
        fi
        prompt='*'
        password+="$char"
    done
    stty echo  # 恢复回显
    echo
    echo "Password: [hidden]"
    ```
