        ```python
            solver.SetTolerance(1e-8)

            # 加载数据并进行处理
            # ... 数据处理代码

        def parallel_toast_processing(data_dict, wavelengths, max_workers=4):
            """
            并行处理多个波长的数据
            """
            results = {}
            
            with concurrent.futures.ThreadPoolExecutor(max_workers=max_workers) as executor:
                # 创建任务字典
        ```

    3. 使用多进程避免GIL限制

        由于TOAST++底层是C++实现，Python的GIL不会影响其计算性能，但如果有大量Python代码或I/O操作，可以使用多进程：

        ```python
        import multiprocessing as mp
        import toast
        import numpy as np

        def process_toast_batch(batch_data, batch_id):
            """
            处理一批数据
            """
            mesh = toast.Mesh()
            solver = toast.Solver(mesh)
            
            # 设置求解器
            solver.SetSolverType('gmres')
            solver.SetTolerance(1e-6)
            solver.SetMaxIter(1000)
            
            results = []
            for data_point in batch_data:
                result = solver.ForwardProject(data_point)
                results.append(result)
            
            return batch_id, np.array(results)

        def parallel_batch_processing(all_data, n_processes=4):
            """
            并行处理多个批次的数据
            """
            # 分割数据
            batch_size = len(all_data) // n_processes
            batches = []
            
            for i in range(n_processes):
                start_idx = i * batch_size
                end_idx = start_idx + batch_size if i < n_processes - 1 else len(all_data)
                batches.append(all_data[start_idx:end_idx])
            
            # 创建进程池
            pool = mp.Pool(processes=n_processes)
            
            # 提交任务
            results = []
            for i, batch in enumerate(batches):
                result = pool.apply_async(process_toast_batch, (batch, i))
                results.append(result)
            
            # 关闭进程池
            pool.close()
            pool.join()
            
            # 收集结果
            final_results = {}
            for result in results:
                batch_id, batch_results = result.get()
                final_results[batch_id] = batch_results
            
            return final_results

        # 使用示例
        if __name__ == "__main__":
            # 准备数据
            all_data = [np.random.rand(100, 50) for _ in range(100)]
            
            # 并行处理
            results = parallel_batch_processing(all_data, n_processes=4)
        ```

    4. 高级并行策略 - 结合OpenMP和Python线程

        ```python
        import os
        import toast
        import numpy as np
        import concurrent.futures
        from dataclasses import dataclass
        from typing import List, Dict, Any

        @dataclass
        class ToastConfig:
            """TOAST配置类"""
            solver_type: str = 'cg'
            tolerance: float = 1e-8
            max_iterations: int = 1000
            preconditioner: str = 'jacobi'
            n_threads: int = 2

        class ToastParallelProcessor:
            """TOAST并行处理器"""
            
            def __init__(self, config: ToastConfig):
                self.config = config
                # 设置全局OpenMP线程数
                os.environ["OMP_NUM_THREADS"] = str(config.n_threads)
            
            def process_dataset(self, datasets: List[Dict[str, Any]], 
                                max_workers: int = 4) -> List[Any]:
                """
                并行处理多个数据集
                """
                results = []
                
                with concurrent.futures.ThreadPoolExecutor(max_workers=max_workers) as executor:
                    # 提交所有任务
                    future_to_dataset = {
                        executor.submit(self._process_single, dataset): i 
                        for i, dataset in enumerate(datasets)
                    }
                    
                    # 监控进度
                    for future in concurrent.futures.as_completed(future_to_dataset):
                        dataset_idx = future_to_dataset[future]
                        try:
                            result = future.result()
                            results.append((dataset_idx, result))
                            print(f"数据集 {dataset_idx} 处理完成 ({len(results)}/{len(datasets)})")
                        except Exception as e:
                            print(f"数据集 {dataset_idx} 处理失败: {e}")
                
                # 按原始顺序排序
                results.sort(key=lambda x: x[0])
                return [r[1] for r in results]
            
            def _process_single(self, dataset: Dict[str, Any]):
                """
                处理单个数据集
                """
                # 为每个线程创建独立的TOAST对象
                mesh = toast.Mesh()
                solver = toast.Solver(mesh)
                
                # 应用配置
                solver.SetSolverType(self.config.solver_type)
                solver.SetTolerance(self.config.tolerance)
                solver.SetMaxIter(self.config.max_iterations)
                solver.SetPreconditioner(self.config.preconditioner)
                
                # 处理数据
                results = {}
                for key, data in dataset.items():
                    if isinstance(data, np.ndarray):
                        result = solver.ForwardProject(data)
                        results[key] = result
                
                return results

        # 使用示例
        if __name__ == "__main__":
            # 配置
            config = ToastConfig(
                solver_type='cg',
                tolerance=1e-8,
                max_iterations=1000,
                n_threads=2
            )
            
            # 创建处理器
            processor = ToastParallelProcessor(config)
            
            # 准备多个数据集
            datasets = [
                {'760nm': np.random.rand(100, 50), '850nm': np.random.rand(100, 50)},
                {'760nm': np.random.rand(100, 50), '850nm': np.random.rand(100, 50)},
                {'760nm': np.random.rand(100, 50), '850nm': np.random.rand(100, 50)},
            ]
            
            # 并行处理
            results = processor.process_dataset(datasets, max_workers=2)
        ```

* python 中，如果 func_2() 的定义在后面，但是我在前面的 func_1() 中就要调用 func_2()，该怎么办

    在 Python 中，如果函数 `func_2` 的定义在后面，而你想在前面的 `func_1` 中调用它，有以下几种解决方案：

    **方法1：调整函数定义顺序（最简单）**

    将 `func_2` 的定义放在 `func_1` 之前：

    ```python
    def func_2():
        print("这是 func_2")
        return "func_2 的结果"

    def func_1():
        result = func_2()  # 现在可以正常调用
        print(f"func_1 调用了 {result}")

    func_1()
    ```

    **方法2：在调用时才执行函数**

    如果必须保持函数定义的顺序，可以在调用 `func_1` 时才执行：

    ```python
    def func_1():
        def inner():
            result = func_2()  # 这时 func_2 已经定义
            print(f"func_1 调用了 {result}")
        return inner()

    def func_2():
        print("这是 func_2")
        return "func_2 的结果"

    # 调用 func_1
    func_1()  # 正常工作
    ```

    **方法3：使用函数作为参数传递**

    ```python
    def func_1(func):
        result = func()  # 调用传入的函数
        print(f"func_1 调用了传入的函数，结果是：{result}")

    def func_2():
        print("这是 func_2")
        return "func_2 的结果"

    # 将 func_2 作为参数传递给 func_1
    func_1(func_2)
    ```

    **方法4：使用全局变量（不推荐）**

    ```python
    def func_1():
        global func_2
        result = func_2()  # 现在可以调用
        print(f"func_1 调用了 {result}")

    def func_2():
        print("这是 func_2")
        return "func_2 的结果"

    func_1()  # 正常工作
    ```

    **为什么会有这个问题？**

    Python 是解释型语言，代码从上到下执行。当执行到 `func_1` 的定义时，`func_2` 还没有被定义，所以会出现 `NameError`。

    **最佳实践**

    推荐使用**方法1**（调整函数定义顺序）或**方法2**（在调用时执行），这样代码更清晰、更易维护。方法3（参数传递）也很常用，特别是在需要灵活性的场景中。

* 所以 python 没有类似 C 中“函数声明”的语法吗？那如果有两个函数互相调用的需求怎么办

    很好的问题！Python 确实**没有**像 C 语言那样的独立函数声明语法。对于相互调用的函数（循环依赖），有以下几种解决方案：

    **场景：两个函数互相调用**

    ```python
    def func_a():
        # 想要调用 func_b
        return func_b() + " 被 func_a 调用"

    def func_b():
        # 想要调用 func_a
        return func_a() + " 被 func_b 调用"
    ```

    **解决方案**

    **方法1：使用条件判断避免无限递归**

    ```python
    def func_a(called_from=None):
        if called_from == 'b':
            return "func_a 被 func_b 调用"
        return func_b(called_from='a') + " -> func_a"

    def func_b(called_from=None):
        if called_from == 'a':
            return "func_b 被 func_a 调用"
        return func_a(called_from='b') + " -> func_b"

    # 测试
    print(func_a())  # func_b 被 func_a 调用 -> func_a
    ```

    **方法2：将其中一个函数定义为内部函数**

    ```python
    def func_a():
        def func_b():
            return func_a() + " 内部"  # 这里调用外部的 func_a
        
        # 在 func_a 内部调用 func_b
        result = func_b()
        return result

    # 注意：这个例子会有无限递归问题，需要加终止条件
    ```

    **方法3：使用类来封装**
    
    ```python
    class MyFunctions:
        def func_a(self):
            return self.func_b() + " -> func_a"
        
        def func_b(self):
            return self.func_a() + " -> func_b"  # 注意：会无限递归

    # 更好的实现：带状态的类
    class Calculator:
        def __init__(self):
            self.call_stack = []
        
        def add(self, x):
            self.call_stack.append('add')
            result = x + self.multiply(x)
            self.call_stack.pop()
            return result
        
        def multiply(self, x):
            self.call_stack.append('multiply')
            if len(self.call_stack) > 2:  # 防止无限递归
                result = x * 2
            else:
                result = x * self.add(x)
            self.call_stack.pop()
            return result

    calc = Calculator()
    print(calc.add(5))  # 正常工作
    ```

    **方法4：使用函数属性或全局字典**

    ```python
    # 方法4a：使用函数属性
    def func_a():
        return func_a.cache['b']() + " 通过 func_a 调用"

    def func_b():
        return func_b.cache['a']() + " 通过 func_b 调用"

    # 设置相互引用
    func_a.cache = {'b': func_b}
    func_b.cache = {'a': func_a}

    # 方法4b：使用字典管理
    functions = {}

    def register(func):
        functions[func.__name__] = func
        return func

    @register
    def func_a():
        return functions['func_b']() + " 通过字典调用"

    @register
    def func_b():
        return functions['func_a']() + " 通过字典调用"
    ```

    **方法5：延迟执行（最常用）**

    ```python
    def func_a(x):
        def execute():
            return func_b(x) + 10
        return execute

    def func_b(x):
        def execute():
            return func_a(x)() + 20  # 注意这里要调用 execute()
        return execute

    # 使用
    result = func_a(5)()  # 需要两次调用
    print(result)
    ```

    **方法6：使用回调函数**

    ```python
    def func_a(callback=None):
        print("func_a 开始执行")
        if callback:
            result = callback()
            print(f"func_a 收到回调结果: {result}")
        return "func_a 完成"

    def func_b():
        print("func_b 开始执行")
        # 调用 func_a，并传入自己作为回调
        return func_a(lambda: "func_b 的回调结果")

    # 调用
    print(func_b())
    ```

    **实际应用示例：简单的表达式求值**

    ```python
    class Expression:
        def __init__(self):
            self.operations = {
                'add': self.add,
                'multiply': self.multiply
            }
        
        def evaluate(self, expr):
            op = expr['op']
            return self.operations[op](expr)
        
        def add(self, expr):
            left = self.evaluate(expr['left'])  # 可能调用 multiply
            right = self.evaluate(expr['right'])
            return left + right
        
        def multiply(self, expr):
            left = self.evaluate(expr['left'])  # 可能调用 add
            right = self.evaluate(expr['right'])
            return left * right

    # 使用
    expr = {
        'op': 'add',
        'left': {'op': 'multiply', 'left': 2, 'right': 3},
        'right': 4
    }

    calc = Expression()
    print(calc.evaluate(expr))  # 输出: 10 (2*3 + 4)
    ```

    **总结**

    Python 虽然没有 C 语言的函数声明，但通过：

    - **类和对象**（最常用）
    - **延迟执行**
    - **回调函数**
    - **全局字典**

    等方式都能优雅地处理函数相互调用的问题。在实际开发中，如果遇到复杂的相互调用，通常意味着需要重新考虑代码设计，可能需要引入类或重新组织代码结构。

* python 似乎也没有 namespace 相关的机制

    Python 实际上有很强大的命名空间（namespace）机制，只是和 C++ 的 `namespace` 关键字不同。Python 的命名空间是通过模块、类和函数作用域来实现的：

    **Python 的命名空间机制**

    1. **模块即命名空间**

        每个 `.py` 文件自动成为一个独立的命名空间：

        ```python
        # math_operations.py
        def add(x, y):
            return x + y

        PI = 3.14159

        # main.py
        import math_operations
        import math_operations as mo  # 可以取别名

        print(math_operations.add(1, 2))  # 通过模块名访问
        print(mo.PI)                       # 通过别名访问
        print(PI)  # 错误！不能直接访问
        ```

    2. **包（Package）—— 嵌套命名空间**
        
        通过目录结构创建嵌套命名空间：

        ```python
        # 目录结构：
        # mypackage/
        # ├── __init__.py
        # ├── math/
        # │   ├── __init__.py
        # │   ├── algebra.py
        # │   └── geometry.py
        # └── utils/
        #     ├── __init__.py
        #     └── strings.py

        # 使用方式：
        import mypackage.math.algebra
        from mypackage.utils import strings
        from mypackage.math.geometry import area

        # 或者用别名
        import mypackage.math.algebra as alg
        ```

    3. **类作为命名空间**

        ```python
        class MathOperations:
            @staticmethod
            def add(x, y):
                return x + y
            
            @staticmethod
            def multiply(x, y):
                return x * y
            
            PI = 3.14159
            E = 2.71828

        class StringOperations:
            @staticmethod
            def capitalize(text):
                return text.capitalize()
            
            @staticmethod
            def reverse(text):
                return text[::-1]

        # 使用
        print(MathOperations.add(1, 2))
        print(MathOperations.PI)
        print(StringOperations.reverse("hello"))
        ```

    4. **函数/方法作用域作为临时命名空间**

        ```python
        def outer_function():
            x = 10  # 在 outer_function 命名空间中
            
            def inner_function():
                y = 20  # 在 inner_function 命名空间中
                return x + y  # 可以访问外部的 x
            
            return inner_function()
        ```

    5. **使用字典模拟命名空间**
        
        ```python
        # 创建命名空间
        math_ns = {
            'add': lambda x, y: x + y,
            'subtract': lambda x, y: x - y,
            'PI': 3.14159
        }

        string_ns = {
            'capitalize': str.capitalize,
            'reverse': lambda s: s[::-1]
        }

        # 使用
        print(math_ns['add'](5, 3))
        print(string_ns['reverse']('hello'))

        # 或者用 SimpleNamespace
        from types import SimpleNamespace

        math = SimpleNamespace(
            add=lambda x, y: x + y,
            subtract=lambda x, y: x - y,
            PI=3.14159
        )

        print(math.add(5, 3))
        print(math.PI)
        ```

    6. **Enum 作为常量命名空间**

        ```python
        from enum import Enum, auto

        class Color(Enum):
            RED = auto()
            GREEN = auto()
            BLUE = auto()
            
        class Direction(Enum):
            NORTH = 'north'
            SOUTH = 'south'
            EAST = 'east'
            WEST = 'west'

        # 使用
        print(Color.RED)
        print(Direction.NORTH.value)
        ```

    7. **动态导入和命名空间**

        ```python
        import importlib

        # 动态导入模块
        module_name = "math"
        math_module = importlib.import_module(module_name)

        # 动态访问属性
        function_name = "sqrt"
        sqrt_func = getattr(math_module, function_name)

        print(sqrt_func(16))  # 4.0
        ```

    **高级命名空间技巧**

    1. **命名空间包（Namespace Packages）**
        
        ```python
        # 不同目录下的同名包会自动合并
        # directory1/
        #   mypackage/
        #     module1.py
        # directory2/
        #   mypackage/
        #     module2.py

        # 使用：
        import mypackage.module1
        import mypackage.module2  # 都能正常工作
        ```

    2. **使用 __all__ 控制导出**
        
        ```python
        # module.py
        __all__ = ['public_function', 'PublicClass']  # 明确导出的内容

        def public_function():
            pass

        def _private_function():  # 约定私有
            pass

        class PublicClass:
            pass

        # 在其他文件
        from module import *  # 只导入 __all__ 中列出的内容
        ```

    3. **动态创建命名空间**
        
        ```python
        def create_namespace(**kwargs):
            """动态创建命名空间"""
            return type('DynamicNamespace', (), kwargs)

        # 使用
        MathNS = create_namespace(
            add=lambda x, y: x + y,
            PI=3.14159
        )

        print(MathNS.add(1, 2))
        print(MathNS.PI)
        ```

    **实际应用示例**

    ```python
    # 一个完整的命名空间示例
    # config/
    # ├── __init__.py
    # ├── database.py
    # └── app.py

    # config/database.py
    class Config:
        HOST = 'localhost'
        PORT = 5432
        USER = 'admin'
        
    class Development(Config):
        DEBUG = True
        DATABASE = 'dev_db'

    class Production(Config):
        DEBUG = False
        DATABASE = 'prod_db'

    # config/app.py
    from .database import Development, Production

    class AppConfig:
        VERSION = '1.0.0'
        NAME = 'MyApp'
        
        class Features:  # 嵌套命名空间
            BETA = False
            ANALYTICS = True
            CACHE_ENABLED = True

    # main.py
    from config.app import AppConfig
    from config.database import Development

    print(AppConfig.VERSION)
    print(AppConfig.Features.BETA)
    print(Development.HOST)
    ```

    **总结**

    Python 虽然没有 `namespace` 关键字，但提供了多种强大的命名空间机制：

    - **模块** - 最基础的命名空间
    - **包** - 模块的集合
    - **类** - 组织和封装相关功能
    - **函数** - 局部作用域
    - **字典/SimpleNamespace** - 动态命名空间

    这些机制共同提供了比 C++ 的 `namespace` 更灵活、更强大的命名空间管理能力。

* 什么是佛教中的名相分别

    在佛教中，“名相分别”是一个重要的概念，它涉及我们对世界的认知方式和修行的关键。我们可以从字面意思和深层含义两个层面来理解：

    **字面意思拆解**

    1.  **名**：指概念、名称、语言文字。它是我们用来指代事物的标签，比如“桌子”、“电脑”、“快乐”。
    2.  **相**：指事物的相状、表象、特征。这是我们通过感官（眼、耳、鼻、舌、身）所接触到的、意识所认识到的各种现象的样子，比如形状、颜色、声音、感觉。
    3.  **分别**：指心识进行区分、辨别、判断的活动。也就是我们运用“名”去认识和界定“相”的过程。

    所以，**“名相分别”合起来，就是指我们凡夫的内心，通过语言文字和概念，去认识和区分世界上各种现象，并在这个过程中产生种种判断和执着。**

    **更深入的理解：佛教为何关注这个概念？**

    在佛教看来，“名相分别”是众生**轮回生死、产生烦恼的根本原因之一**。这是因为：

    1.  **它并非真实的反映，而是主观的构建**：
        佛教认为，万事万物的真实面貌（实相）是超越了“名”和“相”的。但我们习惯于给事物安立名称，并认为这个名称就代表了事物本身。比如，我们看到一个细长、金色、会闪光的东西，给它起名叫“金条”，然后我们对“金条”这个“名”和它的“相”产生了贪爱和执着。但实际上，它只是在一定条件下（温度、形状等）暂时呈现为那个样子的物质元素组合而已。

    2.  **它是二元对立的根源**：
        “分别”本身就是一种分割和比较。有了“长”，就分别出“短”；有了“美”，就分别出“丑”；有了“我的”，就分别出“你的”。这种二元对立的思维模式，正是产生贪爱（想要自己喜欢的）、嗔恨（排斥自己不喜欢的）和愚痴（不明白真相）的基础。

    3.  **它让我们陷入概念之网，无法见到实相**：
        我们活在自己用概念构建的世界里，而不是活在真实的世界里。看到一朵花，我们立刻用“玫瑰”、“美丽”、“红色的”这些“名相”去框定它，然后心念就停留在这些概念上，而不是去直接、鲜活地体验这朵花本身。就像用手指指月亮，我们却执着于手指（名相），而忘记了去看月亮（实相）。

    **修行中的意义**

    佛教修行的目的，不是让我们变成白痴，不认识桌子、椅子，而是为了**超越对“名相”的执着**。

    *   **凡夫的状态**：被“名相分别”所束缚，认为名相就是真实，从而产生执着和痛苦。看到钱就开心，看到批评就生气，完全被这些概念牵着鼻子走。
    *   **修行者的过程**：在修行的初期，我们仍然需要使用“名相”（如“五蕴”、“无常”、“无我”）来学习和理解佛法。这被称为“方便”，是渡河的船筏。
    *   **觉悟者的状态**：彻底看透了“名相”的虚幻性，了知它们只是假名安立，不再被其束缚。他们能使用名相与人沟通，但内心毫不执着。就像《金刚经》里常说的：“佛说第一波罗蜜，即非第一波罗蜜，是名第一波罗蜜。”这就是在指出，我们要通过“名相”去领悟那超越“名相”的真实。

    **总结**

    *   **名相分别**：是我们用概念和标签去认识世界的认知模式。
    *   **它的弊端**：制造二元对立，引发执着和烦恼，遮盖了世界的真实面貌（空性、缘起性）。
    *   **修行的方向**：认识到“名相分别”的局限性，在禅修和生活中，尝试超越概念的束缚，以更直接、更智慧的心去体认实相，最终获得解脱。

    打个比方，就好像你看一场电影，你知道电影里的悲欢离合是情节（相），角色名字是标签（名），但你不会冲上去打银幕上的坏人，因为你内心没有“分别”执着那是真实的。佛教的智慧，就是让我们在人生的这场“电影”中，也能时刻保持这份清醒。

* 简述 python 中 unittest 的用法

    **Python unittest 基本用法**

    unittest 是 Python 的标准单元测试框架，以下是其核心用法：

    1. 基本结构

        ```python
        import unittest

        # 测试类必须继承 unittest.TestCase
        class TestMathFunctions(unittest.TestCase):
            
            # 测试方法必须以 test_ 开头
            def test_add(self):
                self.assertEqual(1 + 1, 2)
                self.assertNotEqual(1 + 1, 3)
            
            def test_subtract(self):
                self.assertTrue(5 - 3 == 2)
                self.assertFalse(5 - 3 == 1)

        # 运行测试
        if __name__ == '__main__':
            unittest.main()
        ```

    2. 常用的断言方法

        ```python
        class TestAssertions(unittest.TestCase):
            
            def test_assertions(self):
                self.assertEqual(2 + 2, 4)              # 相等
                self.assertNotEqual(2 + 2, 5)            # 不相等
                self.assertTrue(True)                     # 为真
                self.assertFalse(False)                   # 为假
                
                self.assertIn(3, [1, 2, 3])               # 包含
                self.assertNotIn(4, [1, 2, 3])            # 不包含
                
                self.assertIs(None, None)                  # 是同一个对象
                self.assertIsNot(None, [])                  # 不是同一个对象
                
                self.assertIsInstance([], list)            # 是某类型
                self.assertNotIsInstance({}, list)         # 不是某类型
                
                self.assertAlmostEqual(3.14159, 3.1416, places=4)  # 浮点数近似相等
                self.assertRaises(ValueError, int, 'abc')  # 预期抛出异常
        ```

    3. 测试固件（setUp 和 tearDown）

        ```python
        class TestDatabase(unittest.TestCase):
            
            def setUp(self):
                """每个测试方法执行前运行"""
                print("准备测试数据...")
                self.data = [1, 2, 3]
            
            def tearDown(self):
                """每个测试方法执行后运行"""
                print("清理测试数据...")
                self.data = None
            
            def test_length(self):
                self.assertEqual(len(self.data), 3)
            
            def test_sum(self):
                self.assertEqual(sum(self.data), 6)
        ```

    4. 类级别的 setUp 和 tearDown

        ```python
        class TestDatabase(unittest.TestCase):
            
            @classmethod
            def setUpClass(cls):
                """整个测试类执行前运行一次"""
                print("连接数据库...")
                cls.connection = "database_connection"
            
            @classmethod
            def tearDownClass(cls):
        ```

    在 Python 环境下处理高模（High-poly）到低模（Low-poly）的简化（Decimation / Simplification），最常用且高效的方案是使用 **PyVista**（基于 VTK）或 **Open3D**。

    以下是几种主流的实现方式：

    1. 使用 PyVista (推荐：功能最强，支持多种算法)

        PyVista 是处理三维网格最直观的库。它内置了 **"Decimate"** 功能，可以精确控制减面的比例。

        ```python
        import pyvista as pv
        import numpy as np

        # 1. 构造 Mesh 对象 (假设你的 nodes 和 faces 是 numpy 数组)
        # 注意：PyVista 的 faces 格式通常需要一个前缀表示点的数量 (例如 [3, 0, 1, 2] 表示三角形)
        # 如果你的 faces 是 (N, 3)，需要转换：
        cells = np.hstack(np.column_stack([np.full(len(mesh_faces), 3), mesh_faces]))
        poly_data = pv.PolyData(mesh_nodes, cells)

        # 2. 执行减面 (减掉 80% 的面，保留 20%)
        # target_reduction 是目标减面率 (0 到 1 之间)
        low_poly = poly_data.decimate(target_reduction=0.8)

        # 3. 导出结果
        low_nodes = low_poly.points
        low_faces = low_poly.faces.reshape(-1, 4)[:, 1:] # 还原回 (N, 3) 格式
        ```

    2. 使用 Open3D (推荐：适合点云与简单网格处理)

        Open3D 的减面算法非常成熟，特别是 **"Quadrric Error Metrics"**（二次误差度量），能在保持外形的同时大幅降低面数。

        ```python
        import open3d as o3d
        import numpy as np

        # 1. 创建网格对象
        mesh = o3d.geometry.TriangleMesh()
        mesh.vertices = o3d.utility.Vector3dVector(mesh_nodes)
        mesh.triangles = o3d.utility.Vector3iVector(mesh_faces)

        # 2. 执行简化
        # target_number_of_triangles 是你希望保留的具体三角形数量
        target_count = len(mesh_faces) // 5  # 简化为原来的 20%
        low_poly = mesh.simplify_quadric_decimation(target_number_of_triangles=target_count)

        # 3. 获取结果
        low_nodes = np.asarray(low_poly.vertices)
        low_faces = np.asarray(low_poly.triangles)

        ```

    3. 如何选择算法？

        | 工具 | 优势 | 适用场景 |
        | --- | --- | --- |
        | **PyVista (Decimate)** | 速度极快，适合超大规模数据。 | 基础地形、快速预览。 |
        | **Open3D (Quadric)** | 形状保持非常好，特征边缘（边缘、尖角）保留更完整。 | 需要精细外观的高模物体。 |
        | **Trimesh** | 纯 Python 友好，集成度高。 | 简单的自动化脚本或轻量化需求。 |

    **进阶建议**

    1. **数据清洗**：在高模减面之前，建议先运行一次 `mesh.remove_duplicated_vertices()` 和 `mesh.remove_degenerate_triangles()`。脏数据会导致减面算法报错或产生空洞。
    2. **法线贴图 (Baking)**：如果你是为了游戏或实时渲染，单纯减面会丢失细节。通常流程是在 Blender 或 Python 中将高模的细节“烘焙”到低模的 **法线贴图 (Normal Map)** 上。
    3. **各向同性重网格化 (Remeshing)**：如果减面后三角形变得过于细长，可以尝试用 `PyVista` 的 `subdivide` 或 `smooth` 进行后处理。

        * basis_inv = toast.Raster(mesh_inv, grd) 这个是干嘛用的

            在 TOAST++ 中，`toast.Raster` 是一个非常关键的**基函数映射器（Basis Mapper）**。

            简单来说，它的作用是建立**有限元网格（Mesh）**与**规则像素网格（Raster Grid）**之间的桥梁。

            1. 为什么要用它？

                在光学断层扫描中，存在两种空间表达方式：

                * **有限元网格 (Mesh)：** 用于数值计算（求解扩散方程）。它通常是不规则的三角形（2D）或四面体（3D），节点分布不均匀，方便适应复杂的几何边界。
                * **规则网格 (Raster/Pixel Grid)：** 用于图像重建和显示。我们习惯把待恢复的参数（如 $\mu_a$）看作一张 100x100 的图片。

                `toast.Raster(mesh_inv, grd)` 创建了一个对象，它知道如何把不规则网格上的物理量“重采样”到规则的像素矩阵上，反之亦然。

            2. 它在代码中的具体功能

                在你的代码中，这个 `basis_inv` 主要承担了以下三个任务：

                **A. 坐标映射与重采样 (M -> B)**

                ```python
                bmua = basis_inv.Map('M->B', mua)

                ```

                * **M (Mesh):** 节点上的矢量（长度等于网格节点数）。
                * **B (Basis):** 像素上的矢量（长度等于 $100 \times 100 = 10,000$）。
                * **作用：** 把有限元节点上的吸收系数插值到 100x100 的图像矩阵中，这样你才能用 `imshow` 把它画出来。

                **B. 降维与参数化 (B -> S)**

                ```python
                scmua = basis_inv.Map('B->S', bcmua)

                ```

                * **S (Solution):** 优化算法实际处理的向量。
                * **作用：** 有时我们不需要在每个网格节点上都求解参数，而是在像素块上求解。这大大减少了逆问题的未知数数量，起到一定的正则化作用。

                **C. 计算灵敏度 (Jacobian)**

                ```python
                J = mesh_inv.Jacobian(basis_inv.Handle(), dphi, aphi, proj)

                ```

                * **核心用途：** 告诉雅可比矩阵计算器：“请计算**每一个像素点**对测量信号的灵敏度，而不是每一个有限元节点”。
                * 这样得到的雅可比矩阵 $J$ 的列数就等于像素总数，使得优化过程直接在“图像层级”进行。

            3. 形象理解

                你可以把它想象成一个**“转换插头”**：

                * 一边插在**复杂的物理网格**上（用于模拟光的传播）。
                * 另一边插在**整齐的显示器**上（用于生成重建图像）。

                **总结：**
                没有这一行，你的程序只能处理枯燥的节点数据矢量；有了它，程序才能处理“图像”，并允许你在规则的 $100 \times 100$ 像素阵列上进行数学优化。

        在不知道真实 $\mu_a, \mu_s$ 的情况下，从一个初始猜想开始：

        这里引入了 **扩散系数 $\kappa = \frac{1}{3(\mu_a + \mu_s)}$**，在 TOAST 中通常对 $\mu_a$ 和 $\kappa$ 进行优化。

        * `bmua = basis_inv.Map('M->B', mua)`

            这部分代码的作用是**将物理参数从“有限元网格”转换到“优化算法所使用的变量空间”**。

            简单来说，重建算法并不是直接在三角形网格的节点上乱试，而是将参数映射到一个更规整、更利于数学计算的“解空间”（Solution Basis）中。

            以下是逐行的详细解释：

            1. 物理场到像素基底的转换 (M -> B)

                ```python
                bmua = basis_inv.Map('M->B', mua)
                bmus = basis_inv.Map('M->B', mus)
                bkap = basis_inv.Map('M->B', kap)
                ```

                * **M (Mesh)**: 指有限元网格节点上的数据。
                * **B (Basis)**: 指像素网格（即之前定义的 100x100 `grd`）上的数据。
                * **作用**: 将定义在不规则网格节点上的吸收系数（$\mu_a$）、散射系数（$\mu_s$）和扩散系数（$\kappa$）插值映射到 100x100 的规则图像像素中。

            2. 物理量单位缩放 (乘以 cm)

                ```python
                bcmua = bmua * cm
                bckap = bkap * cm
                ```

                * **cm**: 是介质中的光速（`c0/refind`）。
                * **作用**: 在 TOAST++ 的底层数学模型中，通常处理的是“变换后的参数”。乘以光速是为了匹配前向方程（Diffusion Equation）中对时间尺度或传播特性的定义，使其符合求解器的数值量级。

            3. 像素基底到解向量的转换 (B -> S)

                ```python
                scmua = basis_inv.Map('B->S', bcmua)
                sckap = basis_inv.Map('B->S', bckap)
                ```

                * **S (Solution)**: 代表“解空间”向量。
                * **作用**: 这是最后一步映射。
                * 在很多情况下，**B** 和 **S** 是等价的（都是像素点）。
                * 但如果使用了特殊的正则化或基函数压缩，`B->S` 会将图像数据压缩成优化算法实际要调整的一维长向量。
                * **scmua** (Solution-space c*mua) 就是算法最终要优化的“未知数”。

            **总结：为什么要绕这么多圈？**

            这几行代码完成了一个**“数据格式标准化”**的过程：

            1. **从几何出发**：把长得像“渔网”一样的节点数据（Mesh），变成长得像“照片”一样的像素数据（Basis）。
            2. **从物理出发**：引入光速 $c_m$，转换成算法内部定义的物理量。
            3. **从数学出发**：将其打包成一个纯数值向量（Solution），交给后面的非线性共轭梯度法（NLCG）去不断迭代寻找最优解。

            **形象比喻：**

            这就像是你要装修房子。

            * **M** 是施工图纸（不规则、复杂）。
            * **B** 是实际的墙面（规整、分块）。
            * **S** 是你的装修预算清单（纯数字列表）。

            这几行代码就是把“图纸上的设计”转化为“预算清单上的数字”，这样你才能用计算器（优化算法）去算怎么省钱。

    5. 非线性共轭梯度 (NLCG) 循环

        这是代码的核心迭代过程：

        A. 计算梯度

        ```python
        dphi = mesh_inv.Fields(...) # 前向场
        aphi = mesh_inv.Fields(...) # 伴随场
        # 计算雅可比矩阵 (Jacobian/Sensitivity Matrix)
        J = mesh_inv.Jacobian(basis_inv.Handle(), dphi, aphi, proj)
        # 链式法则计算目标函数关于 logx 的梯度 r
        r = matrix(J).transpose() * (2*(data-proj)/sd**2)
        r = np.multiply(r, x) # 对应对数变换的导数
        ```

        通过 **伴随场方法 (Adjoint Method)** 高效计算雅可比矩阵，进而得到梯度方向。

        B. 更新搜索方向 (CG 逻辑)

        ```python
        if itr == 1:
            d = s # 第一步使用最速下降方向
        else:
            beta = (delta_new - delta_mid) / delta_old # Polak-Ribière 公式计算 beta
            d = s + d * beta # 构造共轭方向

        ```

        这里利用前几次迭代的信息来修正搜索方向，使收敛速度快于普通的最速下降法。

        C. 线搜索与参数更新

        ```python
        # 沿着方向 d 寻找最优步长 step
        step, err = toast.Linesearch(logx, d, step, err, objective_ls)
        logx = logx + d * step # 更新解

        ```

        `toast.Linesearch` 确保每一步迭代都能有效降低目标函数值。

* 为什么要用户手动计算非线性共轭梯度循环？没有省略细节，一步到位的方法吗？

    这是一个非常深刻的问题。在深度学习时代，我们习惯了 `model.fit()` 或 `loss.backward()` 这种“黑盒”操作，但 **TOAST++** 作为一个科研级的偏微分方程（PDE）反演框架，故意把这些细节暴露给用户。

    之所以要手动写这个循环，主要有以下三个原因：

    1. 物理反演的复杂性：非线性与约束

        与简单的机器学习不同，光学断层扫描（DOT）是一个**高度非线性的病态问题**。

        * **参数约束**：物理参数（如 $\mu_a, \kappa$）必须为正数。代码中通过 `logx = np.log(x)` 将问题转入对数域，这在通用库里很难自动实现。
        * **多参数耦合**：你同时在重建两个物理场（吸收和散射）。手动循环允许你对不同的参数设置不同的权重或正则化惩罚。

    2. 计算开销的权衡

        在 DOT 中，每一步迭代最耗时的不是梯度更新，而是**前向模型（Forward Model）**的求解。

        * 手动写循环可以让你精确控制什么时候调用 `mesh_inv.Fields`（求解 PDE）。
        * 你可以自定义**线搜索（Line Search）**的策略。如果步长选得不好，物理计算可能会发散，手动控制能提高鲁棒性。

    3. TOAST++ 的设计哲学

        TOAST++ 定位为一个**工具箱（Toolkit）**，而不是一个**黑盒软件**。它提供了最难写的“零件”：

        * 高性能的有限元求解器（FEM Solver）
        * 高效的伴随场梯度计算（Jacobian）
        * 网格与基底的映射（Mapper）

        至于如何利用这些零件去“跑”优化，它把控制权交给了科研人员，方便你随时加入自定义的**正则化项（Regularization）**、**空间先验信息**或**多波长约束**。

    **有没有“一步到位”的方法？**

    如果你觉得手动写 NLCG 太麻烦，确实有几种简化方案：

    方案 A：使用 SciPy 的优化库 (推荐)

    你可以把前向求解和梯度计算封装成一个 Python 函数，直接丢给 `scipy.optimize.minimize`。这样你就不用写 beta 的计算和线搜索逻辑了：

    ```python
    from scipy.optimize import minimize

    def func(logx):
        # 这里写前向计算和目标函数
        return f, grad

    # 一行代码代替整个 while 循环
    res = minimize(func, logx0, method='L-BFGS-B', jac=True)

    ```

    * **优点**：省去了写 CG 逻辑的麻烦，且 L-BFGS 通常比 NLCG 收敛更快。
    * **缺点**：在大规模 3D 重建中，SciPy 的内存管理可能不如手动控制精细。

    方案 B：使用更高层的封装（如 PyTorch）

    现在有些研究者会将 TOAST 的前向算子包装成 **PyTorch Autograd Function**。

    * **优点**：你可以直接用 `optimizer.step()`，甚至结合神经网络（深度先验）。
    * **现状**：这需要你自己写一个桥接层（Wrapper），目前 TOAST++ 官方示例仍倾向于展示底层的数值过程。

    **总结**

    手动写循环虽然看起来“原始”，但在科研中它提供了**极致的透明度**。当你发现重建图像有伪影时，你可以直接在循环里检查是梯度方向错了，还是步长选得太大。

        * 为了求组织分布，你需要知道光路 (PMDF)。

        * 为了知道光路 (PMDF)，你必须先知道组织分布。

    这就是为什么你不能像普通公式那样一步算出答案。

    再看你之前的代码逻辑，它其实是在玩一个“猜谜游戏”：

        第一步（瞎猜）：假设组织是均匀的（初始 mua=0.025）。

        第二步（算光路）：基于这个均匀的假设，算出一个“香蕉形”的光路（计算 Jacobian）。

        第三步（修正）：用这个光路去对比实际测量数据，发现不对劲，于是微调一下图像。

        第四步（更新光路）：既然图像变了，光路肯定也变了！于是重新计算 Jacobian。

        循环：重复以上步骤，直到图像和光路互相吻合。

    * 非线性：因为“光路会随组织变化而扭曲”，所以必须一边改图像，一边改光路（迭代）。

    * Gmsh (.msh)：这是目前科学研究中最通用的开源网格格式。它的文件头通常是 $MeshFormat。

    ```
    MeshData 5.0

    NodeList 3511 1
    N[24.6056 0.429493]R0
    N[24.5756 1.28796]R0 
    ...
    ```

    1. 文件头信息

        MeshData 5.0: 文件格式版本号。

        NodeList 3511 1:

            3511: 代表这个网格总共有 3511 个节点。

            1: 代表数据的列数（通常用于标识数据维度或坐标系类型，这里指二维空间坐标）。这个 1 并不是指坐标的列数（X 和 Y），而是指附加在节点上的标量数据列数。

            1：表示除了坐标外，每个节点还携带了 1 组 额外的数据（即你看到的那个 R0 中的 0）。

    2. 节点坐标数据 (Node Data)

        每一行代表一个节点的信息，格式通常为：N[X坐标 Y坐标]R区域标识。

        以第一行为例：N[24.6056 0.429493]R0

            N: 标识这是一个节点（Node）。

            [24.6056 0.429493]: 这是该节点在二维坐标系中的位置。

                X = 24.6056

                Y = 0.429493

            R0: R 代表 Region（区域）。0 表示这个节点属于编号为 0 的区域。在复杂的网格中，不同的 R 值可以区分不同的组织（例如 R0 是皮肤，R1 是大脑）。

    3. 两种 nodes

        在这种上下文中，N 和 B 通常代表以下含义：

        1. N (Node / Internal Node)：内部节点

            含义：标准的网格节点，位于模型内部或一般的几何位置。

            坐标：后面括号内的是其几何坐标，例如 N[24.6056 0.429493] 表示该点的 X 为 24.6056，Y 为 0.429493。

            状态：通常表示这是一个自由节点，或者是不在边界约束上的普通点。

        2. B (Boundary Node)：边界节点

            含义：B 代表 Boundary（边界）。这些是位于几何模型边缘、轮廓或特定边界条件线上的节点。

            为什么区分：

                边界条件赋值：在仿真计算中，软件需要快速识别哪些点在边界上，以便施加压力、位移约束（Fixed）或流量。

                网格重构：在自动划分网格时，边界点通常是固定的（Constraints），不能像内部点那样轻易移动位置。

            数据结构：虽然开头字母不同，但 B 后面的数据格式 [24.9848 0.872487] 与 N 完全一致，说明它们在几何本质上都是点（Point/Vertex）。

        ```
        Size 3511
        Param1 MUA
        Param2 KAPPA
        Param3 N
        Data
        0.01 0.330033 1.4
        0.01 0.330033 1.4
        0.01 0.330033 1.4
        ...
        ```

        在网格数据（Mesh Data）上下文中，它表示为你之前提到的 3511 个节点（Nodes） 每一项分配的物理属性或材料参数。

        * Size 3511: 这里的数量与你之前的 NodeList 数量完全一致（都是 3511）。这意味着每一行数据对应一个节点。

        * Param1/2/3: 这些是物理参数的名称。

        * MUA ($μ-a$​): 通常在生物光子学或输运理论中代表 吸收系数 (Absorption Coefficient)。

        * KAPPA ($kappa$): 通常代表 扩散系数 (Diffusion Coefficient) 或 热导率。

        * N ($n$): 通常代表 折射率 (Refractive Index)。

        * Data: 下方的三列数字分别对应这三个参数。

    5. boundary data

        根据你提供的 `surf_data` 结构，这是一个典型的 **边界/表面描述数据**。它通常用于定义网格的“外壳”或者特定的边界线段。

        在有限元或计算几何中，这通常被称为 **Boundary Mesh** 或 **Surface Elements**。

        我们可以从这三个 `array` 的特征来拆解它的含义：

        1. 第一个数组：`nodes_coords` (N, 2)

            * **内容**：`[[24.9848, 0.872487], ...]`
            * **含义**：这是**边界节点的几何坐标**。
            * **关联**：你会发现这些坐标与你之前提到的 `B[...]`（Boundary Nodes）是一一对应的。它们定义了物体最外层的形状。

        2. 第二个数组：`edges/lines` (M, 2)

            * **内容**：`[[89, 95], [134, 89], ...]`，类型为 `int32`。
            * **含义**：这是**拓扑连接关系（边）**。
            * 每一行代表一根线段。例如 `[89, 95]` 表示第 89 号节点和第 95 号节点之间连接成一根线。
            * 因为每行只有 2 个数字，说明这是 **1D 单元（线段）**，而不是三角形面。

            * **作用**：它定义了网格的**轮廓线**。在仿真中，这些边通常用来施加边界条件（如：这圈线是光源照射的位置，或者是绝热边界）。

        3. 第三个数组：`node_indices` (K,)

            * **内容**：`[360, 361, ..., 3510]`，类型为 `int32`。
            * **含义**：这是**全局索引映射**。
            * **作用**：它告诉软件：这组 `surf_data` 中的第 0 号点，实际上对应于主 `NodeList`（全网格节点列表）里的第 360 号点。这是一种“局部到全局”的映射关系。

        **总结：这组数据是做什么用的？**

        这份 `surf_data` 的完整定义是：**该网格的外部边界轮廓。**

        * 它由一系列线段（Edges）组成。
        * 它只包含边界上的点（也就是你之前问的 `B` 点）。
        * 它主要用于：

            1. **绘图**：在画出密密麻麻的三角形网格后，用粗黑线勾勒出物体的外轮廓。
            
            2. **物理计算**：定义光线从哪些边射入（Source），或者从哪些边射出（Detector）。

        **对比建议：**

        * 如果你用 `faces (M, 3)` 画图，你得到的是一个**实心的面**。
        * 如果你用这个 `surf_data` 的 `edges (M, 2)` 画图，你得到的是一圈**空心的线条**。

    1. 模块导入与环境准备

        ```python
        import os
        import sys
        import numpy as np
        from numpy import matrix
        from scipy import sparse
        from scipy.sparse import linalg
        import matplotlib.pyplot as plt

        from toast.mesh import Mesh # 导入 TOAST 核心网格类

        ```

        * 导入必要的数值计算库。`scipy.sparse.linalg` 用于求解大型稀疏线性方程组（因为有限元矩阵非常稀疏）。

    2. 加载网格与实验配置

        ```python
        meshdir = os.path.expandvars("$TOASTDIR/test/2D/meshes/") # 获取测试网格路径
        meshfile = meshdir + "circle25_32.msh" # 2D圆形有限元网格文件
        qmfile = meshdir + "circle25_32x32.qm" # 定义了 32 个光源和 32 个检测器的位置

        mesh = Mesh()
        mesh.Read(meshfile) # 加载几何网格
        mesh.ReadQM(qmfile)  # 加载光源(Q)和检测器(M)的配置

        ```

        * `Mesh` 对象是 TOAST 的核心，它存储了所有几何信息。
        * `QM` 文件决定了光从哪里进去，从哪里出来。

    3. 定义物理参数

        ```python
        nlist,elist,eltp = mesh.Data() # 获取节点列表、单元列表等原始数据
        nlen = nlist.shape[0]          # 获取网格节点总数（此处为 3511）

        # 假设介质是均匀的（Homogeneous）
        mua = np.ones ((1,nlen)) * 0.025 # 吸收系数：全场统一为 0.025 mm^-1
        mus = np.ones ((1,nlen)) * 2.0   # 散射系数：全场统一为 2.0 mm^-1
        ref = np.ones ((1,nlen)) * 1.4   # 折射率：全场统一为 1.4
        freq = 100                       # 调制频率：100 MHz (用于频域测量)

        ```

        * 这一步是在为每个网格节点分配物理属性。

    4. 构建系统方程

        ```python
        smat = mesh.Sysmat(mua, mus, ref, freq) # 生成系统矩阵 (K)
        qvec = mesh.Qvec()                      # 生成光源向量 (Q)
        mvec = mesh.Mvec()                      # 生成检测器映射向量 (M)
        mvec = mvec.transpose()
        ```

        * **`smat`**: 这是扩散方程离散化后的左手边矩阵 $K$。求解方程即求解 $K\Phi = Q$。
        * **`qvec`**: 每一个光源在网格节点上产生的初始分布。
        * **`mvec`**: 定义了如何从全场解 $\Phi$ 中提取边界检测点的信号。

        * SourceList 32 → 对应 phi 的 32 列（32 次独立的前向计算）。

        * MeasurementList 32 → 对应 mvec 的 32 列（32 个提取信号的权重向量）。

    5. 求解线性方程组 (核心计算)

        ```python
        nq = qvec.shape[1] # 获取光源数量 (32)
        phi = np.zeros((nlen,nq),dtype=np.cdouble) # 准备存储全场解的复数矩阵

        for q in range(nq): # 遍历每一个光源
            qq = qvec[:,q].todense() # 取出第 q 个光源的列向量
            # 使用迭代法（双共轭梯度稳定法）求解 K*phi = Q
            res = linalg.bicgstab(smat, qq, rtol=1e-12) 
            phi[:,q] = res[0] # 存储解（光场分布）
        ```

        * 这一步计算量最大。它计算了每一个光源开启时，光在整个圆形组织内部的分布情况。

        * `phi` 的维度是 `(节点数, 光源数)`。

    6. 边界投影与对数变换

        ```python
        y = mvec.transpose() * phi # 核心投影：将全场解映射到边界检测器
        logy = np.log(y)           # 对复数测量值取自然对数

        ```

        * **关键点**：`mvec.transpose() * phi` 的结果是一个 $32 \times 32$ 的矩阵。
        * $y_{ij}$ 表示第 $j$ 个光源在第 $i$ 个检测器收到的信号。


        * **复数对数**：
        * `logy.real` (实部) = $\ln(\text{Amplitude})$，即振幅的衰减。
        * `logy.imag` (虚部) = $\text{Phase}$，即光波的相位滞后。

    7. 结果可视化 (正弦图)

        ```python
        plt.figure(1)
        im = plt.imshow(logy.real,interpolation='none') # 绘制对数振幅图
        plt.title('log amplitude')
        ...
        plt.figure(2)
        im = plt.imshow(logy.imag,interpolation='none') # 绘制相位图
        plt.title('phase')
        ...
        plt.show()

        ```

        * 这两张图被称为 **正弦图 (Sinogram)**。

        Dimension 2：确认坐标维度是 2（x,y）。如果是 3D 实验，这里会是 Dimension 3。

    2. 光源列表 (SourceList)

        ```

        ```

        32：表示总共有 32 个光源。

        fixed：这是一个关键参数。它告诉 TOAST++ 这些光源的位置是固定在特定坐标上的。

            在有限元（FEM）计算时，程序会自动寻找距离这些坐标最近的网格节点，或者在这些坐标周围分配源权重（如 Gaussian 分布）。

        文件中列出的每一行都是一个光源的 (x,y) 坐标：

        (24.5, 0)

        (24.029239, 4.7797129)

        (22.635049, 9.3757441)

    4. 探测器列表 MeasurementList

        MeasurementList 32

            32：表示共有 32 个探测点。

            省略了 fixed：在 TOAST++ 格式中，如果第一行没有写 fixed，它通常默认也是按照坐标读取。但在有些版本里，如果后面直接跟坐标，它就默认为固定位置。

    5. LinkList（连接列表）定义了**激励源（Sources）与测量点（Measurements）**之间的对应关系。简单来说，它决定了“当哪一个发射器工作时，哪些接收器负责采集数据”。

        1. 结构分解

            文件的 LinkList 部分包含 32 行数据，每行的格式如下：

            32: 0 1 2 ... 31

            * 冒号前的数字 (32)： 表示该激励源触发时，后面跟着的测量点总数。

            * 冒号后的索引 (0-31)： 对应 MeasurementList 中测量点的编号。

        2. 具体含义：全通道扫描

            在你的这个文件中，LinkList 的逻辑非常直接且对称：

            * 全通模式： 每一行都包含了从 0 到 31 的所有索引。这意味着无论哪一个 Source 处于活动状态，系统都会同时读取全部 32 个测量点的数据。

            * 数据矩阵： 这种配置会产生一个 32×32 的完整数据集（共 1024 个数据点）。

* 排序链表

    * 题目

        给你链表的头结点 head ，请将其按 升序 排列并返回 排序后的链表 。

        

        示例 1：

        输入：head = [4,2,1,3]
        输出：[1,2,3,4]

        示例 2：

        输入：head = [-1,5,3,4,0]
        输出：[-1,0,3,4,5]

        示例 3：

        输入：head = []
        输出：[]

        

        提示：

            链表中节点的数目在范围 [0, 5 * 104] 内
            -105 <= Node.val <= 105

        

        进阶：你可以在 O(n log n) 时间复杂度和常数级空间复杂度下，对链表进行排序吗？

    * 我的解法

        ```cpp
        /**
        * Definition for singly-linked list.
        * struct ListNode {
        *     int val;
        *     ListNode *next;
        *     ListNode() : val(0), next(nullptr) {}
        *     ListNode(int x) : val(x), next(nullptr) {}
        *     ListNode(int x, ListNode *next) : val(x), next(next) {}
        * };
        */
        class Solution {
        public:
            ListNode* merge(ListNode *list1, ListNode *list2) {
                ListNode *p1 = list1, *p2 = list2;
                ListNode dummy;
                ListNode *dummy_head = &dummy;
                ListNode *p = dummy_head;
                while (p1 && p2) {
        ```

        这个过程就像是在**织布**：每次剪下一段，理顺，缝好，再剪下一段。

        **自底向上归并排序 (Bottom-up Merge Sort)**

        这种方法的难点在于：**你需要手动维护四个关键点**：

        1. **已排序部分的末尾**（用于连接下一段）。
        2. **当前待处理部分的起点**。
        3. **第一段子链表**。
        4. **第二段子链表**。

        ```cpp
        class Solution {
        public:
            ListNode* sortList(ListNode* head) {
                if (!head || !head->next) return head;

                // 1. 先统计链表总长度
                int length = 0;
                ListNode* curr = head;
                while (curr) {
                    length++;
                    curr = curr->next;
                }

                // 2. 引入 dummyHead 简化头部的处理
                ListNode dummy(0);
                dummy.next = head;

                // 3. 循环步长：1, 2, 4, 8...
                for (int step = 1; step < length; step <<= 1) {
                    ListNode* prev = &dummy;     // 已合并部分的末尾
                    curr = dummy.next;           // 剩余未合并部分的起点

                    while (curr) {
                        // --- 找到第一段 ---
                        ListNode* head1 = curr;
                        for (int i = 1; i < step && curr->next; ++i) curr = curr->next;
                        
                        // --- 找到第二段 ---
                        ListNode* head2 = curr->next;
                        curr->next = nullptr;    // 切断第一段
                        curr = head2;
                        for (int i = 1; i < step && curr && curr->next; ++i) curr = curr->next;

                        // --- 切断第二段并保留后续的起点 ---
                        ListNode* next_start = nullptr;
                        if (curr) {
                            next_start = curr->next;
                            curr->next = nullptr; // 切断第二段
                        }

                        // --- 合并这两段并将结果挂到 prev 后面 ---
                        prev->next = merge(head1, head2);
                        
                        // --- 更新 prev 到已合并部分的末尾 ---
                        while (prev->next) prev = prev->next;
                        
                        // --- 移动到下一组待处理位置 ---
                        curr = next_start;
                    }
                }
                return dummy.next;
            }

            // 标准的双链表合并函数
            ListNode* merge(ListNode* l1, ListNode* l2) {
                ListNode dummy(0);
                ListNode* p = &dummy;
                while (l1 && l2) {
                    if (l1->val <= l2->val) {
                        p->next = l1; l1 = l1->next;
                    } else {
                        p->next = l2; l2 = l2->next;
                    }
                    p = p->next;
                }
                p->next = l1 ? l1 : l2;
                return dummy.next;
            }
        };
        ```
